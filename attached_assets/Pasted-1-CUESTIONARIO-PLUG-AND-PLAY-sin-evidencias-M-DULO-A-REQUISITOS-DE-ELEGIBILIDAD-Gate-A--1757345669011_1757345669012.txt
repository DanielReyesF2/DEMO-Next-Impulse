function scoreModule(module, answers) {
  const sum = module.questions.reduce((acc,q)=>{
    const v = q.options[answers[q.id]] ?? 0;
    return acc + v;
  }, 0);
  return (sum / module.questions.length) * module.weight; // 0..module.weight
}

function computeReadiness(modules, answers) {
  // Gate
  const gate = modules.find(m => m.id === "A");
  const gateOk = gate.gate_blockers.every(id => (gate.questions.find(q=>q.id===id).options[answers[id]] ?? 0) >= gate.gate_threshold);
  const readiness = modules
    .filter(m => m.weight > 0)
    .reduce((acc,m)=> acc + scoreModule(m, answers), 0); // 0..1
  return { gateOk, readinessIndex: Math.round(readiness*100) };
}
